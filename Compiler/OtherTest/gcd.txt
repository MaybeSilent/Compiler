int gcd(int a,int b) {
    if(b==0) return(a);
    return(gcd(b,a-a/b*b));
}

void main(){
    int i , a , b;
    int flg;
    flg=1;
    i = 0;
    do{
        printf("please input 2 integers, and the program will work out their gcd.");
        printf("input 2 zeros to quit the loop.");
        scanf(a,b);
        if(a==0) if(b==0) flg=0;
        if(flg) printf("the gcd is: ",gcd(a,b));
        i = i + 1;
    }while(flg)
    i = 10;
}
